180 static void ValidateEntities(EntityData data)
181 {
182 foreach (var entity in data.Entities.Values)
183 {
184 var duplicateField = entity.Fields.Keys
185 .GroupBy(name => name)
186 .FirstOrDefault(g => g.Count() > 1);
187
188 if (duplicateField != null)
189 {
190 throw new Exception($"[Semantic Error]: Entity '{entity.Name}' has multiple fields named '{duplicateField.Key}'.");
191 }
192
193 var duplicateEvent = entity.Events
194 .GroupBy(e => e.Name)
195 .FirstOrDefault(g => g.Count() > 1);
196
197 if (duplicateEvent != null)
198 {
199 throw new Exception($"[Semantic Error]: Entity '{entity.Name}' has multiple definitions for event '{duplicateEvent.Key}'. Event names must be unique.");
200 }
201 }

204 static void ReloadLogic(string path, EntityData currentData)
205 {
206 try
207 {
208 Console.WriteLine("\n[Hot Reload] Changes detected! Processing...");
209 string code = ResolveImports(path, new HashSet<string>());
210
211 EntityData newData = MorphynParser.ParseFile(code);
212
213 foreach (var newEntry in newData.Entities)
214 {
215 string name = newEntry.Key;
216 Entity newEntity = newEntry.Value;
217
218 if (currentData.Entities.TryGetValue(name, out var existingEntity))
219 {
220 existingEntity.Events = newEntity.Events;
221 existingEntity.BuildCache();
222 Console.WriteLine($"[Hot Reload] Logic updated: {name}");
223 }
224 else
225 {
226 newEntity.BuildCache();
227 currentData.Entities.Add(name, newEntity);
228 Console.WriteLine($"[Hot Reload] New entity spawned: {name}");
229
230 if (newEntity.Events.Any(e => e.Name == "init"))
231 {
232 MorphynRuntime.Send(newEntity, "init");
233 }
234 }
235 }
236 }
237 catch (Exception ex)
238 {
239 Console.WriteLine($"[Hot Reload Error]: {ex.Message}");

243 /**
244 * \brief Resolves import statements recursively
245 * \param filePath Path to the file to process
246 * \param visited Set of already processed files (prevents circular imports)
247 * \return Combined content of all imported files
248 *
249 * \page imports Import System
250 *
251 * \section import_syntax Import Syntax
252 *
253 * Import other Morphyn files:
254 *
255 * \code{.morphyn}
256 * import "enemies.morphyn";
257 * import "weapons.morphyn";
258 * import "items.morphyn";
259 *
260 * entity Player {
261 * # Can use entities from imported files
262 * }
263 * \endcode
264 *
265 * \section import_rules Import Rules
266 *
267 * - Imports are resolved relative to the importing file
268 * - Circular imports are automatically prevented
269 * - Import statements must end with semicolon
270 * - Missing import files generate warnings but don't stop execution
271 */
272 static string ResolveImports(string filePath, HashSet<string> visited)
273 {
274 string absolutePath = Path.GetFullPath(filePath);
275 if (visited.Contains(absolutePath)) return "";
276 visited.Add(absolutePath);
277
278 string content = File.ReadAllText(absolutePath);
279 var lines = content.Split('\n');
280 var finalContent = new List<string>();
281
282 foreach (var line in lines)
283 {
284 var trimmed = line.Trim();
285 if (trimmed.StartsWith("import ") && trimmed.EndsWith(";"))
286 {
287 string fileName = trimmed.Replace("import", "").Replace("\"", "").Replace(";", "").Trim();
288 string subPath = Path.Combine(Path.GetDirectoryName(absolutePath)!, fileName);
289
290 if (File.Exists(subPath))
291 finalContent.Add(ResolveImports(subPath, visited));
292 else
293 Console.WriteLine($"[Warning] Import file not found: {subPath}");
294 }
295 else
296 {
297 finalContent.Add(line);
298 }
299 }
300 return string.Join("\n", finalContent);
301 }
302 }
