51 // Terms are the lowest level of the expression.
52 // They include numbers, identifiers, and subexpressions in parentheses.
53 // Subexpressions have higher priority than numbers and identifiers.
54 private static TokenListParser<MorphynToken, MorphynExpression> Term =>
55 Token.EqualTo(MorphynToken.Double).Select(t => (MorphynExpression)new LiteralExpression(double.Parse(t.ToStringValue(), CultureInfo.InvariantCulture)))
56 .Or(Token.EqualTo(MorphynToken.Number).Select(t => (MorphynExpression)new LiteralExpression(double.Parse(t.ToStringValue(), CultureInfo.InvariantCulture))))
57 .Or(Token.EqualTo(MorphynToken.String).Select(t => (MorphynExpression)new LiteralExpression(t.ToStringValue().Trim('"'))))
58 .Or(Token.EqualTo(MorphynToken.True).Select(_ => (MorphynExpression)new LiteralExpression(true)))
59 .Or(Token.EqualTo(MorphynToken.False).Select(_ => (MorphynExpression)new LiteralExpression(false)))
60 .Or(Token.EqualTo(MorphynToken.Null).Select(_ => (MorphynExpression)new LiteralExpression(null)))
61 .Or(IndexAccess)
62 .Or(PropertyAccess)
63 .Or(Identifier.Select(id => (MorphynExpression)new VariableExpression(id)))
64 .Or(Parse.Ref(() => Expression).Between(Token.EqualTo(MorphynToken.LeftParen), Token.EqualTo(MorphynToken.RightParen)));
65
66 private static TokenListParser<MorphynToken, MorphynExpression> Unary =>
67 (from minus in Token.EqualTo(MorphynToken.Minus)
68 from term in Term
69 select (MorphynExpression)new BinaryExpression("-", new LiteralExpression(0), term))
70 .Or(Term);

72 // Parser precedence for multiplication and division (and modulo).
73 // These operators have higher precedence than addition and subtraction.
74 private static TokenListParser<MorphynToken, MorphynExpression> Factor =>
75 Parse.Chain(
76 Token.EqualTo(MorphynToken.Asterisk).Or(Token.EqualTo(MorphynToken.Slash)).Or(Token.EqualTo(MorphynToken.Percent)),
77 Unary,
78 (op, left, right) => (MorphynExpression)new BinaryExpression(op.ToStringValue(), left, right));

80 // Parser precedence for addition and subtraction.
81 // These operators have lower precedence than multiplication and division.
82 private static TokenListParser<MorphynToken, MorphynExpression> Expression =>
83 Parse.Chain(Token.EqualTo(MorphynToken.Or), AndParser,
84 (op, left, right) => (MorphynExpression)new BinaryLogicExpression(left, "or", right));

86 /// <summary>
87 /// An arrow "->" is used in flow actions to indicate that the expression on the left should be assigned to the field on the right.
88 /// For example, "entity.health -> player.health" sets the health of the entity to the health of the player.
89 /// </summary>
90 private static TokenListParser<MorphynToken, MorphynAction> FlowAction =>
91 (from valExpr in Expression
92 from arrow in Token.EqualTo(MorphynToken.Arrow)
93 from poolName in Identifier
94 from dot in Token.EqualTo(MorphynToken.Dot).Optional()
95 from member in Token.EqualTo(MorphynToken.Identifier).Where(t => t.ToStringValue() == "at").Optional()
96 from indexExpr in Expression.Between(Token.EqualTo(MorphynToken.LeftBracket), Token.EqualTo(MorphynToken.RightBracket))
97 select (MorphynAction)new SetIndexAction {
98 TargetPoolName = poolName,
99 IndexExpr = indexExpr,
100 ValueExpr = valExpr
101 }).Try()
102 .Or(from expr in Expression
103 from arrow in Token.EqualTo(MorphynToken.Arrow)
104 from target in Identifier
105 select (MorphynAction)new SetAction { Expression = expr, TargetField = target });

107 /// <summary>
108 /// Flow actions consist of an expression followed by an arrow and an identifier.
109 /// The expression is evaluated and the result is assigned to the field indicated by the identifier.
110 /// </summary>
111 private static TokenListParser<MorphynToken, object> LiteralValue =>
112 Token.EqualTo(MorphynToken.Double)
113 .Select(t => (object)double.Parse(t.ToStringValue(), CultureInfo.InvariantCulture))
114 .Or(Token.EqualTo(MorphynToken.Number).Select(t =>
115 (object)double.Parse(t.ToStringValue(), CultureInfo.InvariantCulture)))
116 .Or(Token.EqualTo(MorphynToken.String).Select(t => (object)t.ToStringValue().Trim('"')))
117 .Or(Token.EqualTo(MorphynToken.True).Select(_ => (object)true))
118 .Or(Token.EqualTo(MorphynToken.False).Select(_ => (object)false))
119 .Or(Token.EqualTo(MorphynToken.Null).Select(_ => (object)null));

255 // Parse entity: entity Name { fields... events... }
256 private static TokenListParser<MorphynToken, Entity> EntityDeclaration =>
257 from entityKeyword in Token.EqualTo(MorphynToken.Entity)
258 from name in Identifier
259 from leftBrace in Token.EqualTo(MorphynToken.LeftBrace)
260 from members in (
261 FieldDeclaration.Select(f => (object)f)
262 .Or(EventDeclaration.Select(e => (object)e))
263 ).Many()
264 from rightBrace in Token.EqualTo(MorphynToken.RightBrace)
265 select new Entity
266 {
267 Name = name,
268 Fields = members.OfType<KeyValuePair<string, object>>()
269 .GroupBy(f => f.Key).Any(g => g.Count() > 1)
270 ? throw new Exception($"[Semantic Error]: Entity '{name}' has duplicate fields.")
271 : members.OfType<KeyValuePair<string, object>>().ToDictionary(f => f.Key, f => f.Value),
272
273 Events = members.OfType<Event>()
274 .GroupBy(e => e.Name).Any(g => g.Count() > 1)
275 ? throw new Exception($"[Semantic Error]: Entity '{name}' has duplicate events.")
276 : members.OfType<Event>().ToList()
277 };

279 // Parse multiple entities
280 private static TokenListParser<MorphynToken, Entity[]> RootParser =>
281 EntityDeclaration.Many().Select(entities =>
282 {
283 var duplicate = entities.GroupBy(e => e.Name).FirstOrDefault(g => g.Count() > 1);
284 if (duplicate != null)
285 throw new Exception($"[Semantic Error]: Duplicate entity definition: '{duplicate.Key}'");
286 return entities.ToArray();
