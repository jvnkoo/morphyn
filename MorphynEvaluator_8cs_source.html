127 public static object? EvaluateExpression(Entity entity, MorphynExpression expr,
128 Dictionary<string, object?> localScope, EntityData data)
129 {
130 return expr switch
131 {
132 LiteralExpression l => l.Value,
133
134 VariableExpression v =>
135 localScope.TryGetValue(v.Name, out var argVal) ? argVal :
136 (entity.Fields.TryGetValue(v.Name, out var fieldVal)
137 ? fieldVal
138 : throw new Exception($"Variable '{v.Name}' not found in '{entity.Name}'")),
139
140 BinaryExpression b => EvaluateBinary(entity, b, localScope, data),
141
142 BinaryLogicExpression bl => EvaluateLogic(entity, bl, localScope, data),
143
144 UnaryLogicExpression u => EvaluateUnary(entity, u, localScope, data),
145
146 IndexAccessExpression idx => GetFromPool(entity, idx, localScope, data),
147
148 PoolPropertyExpression p => GetPoolProperty(entity, p, data),
149
150 _ => throw new Exception($"Unsupported expression: {expr.GetType().Name}")
151 };
152 }
153
154 private static bool EvaluateLogic(Entity entity, BinaryLogicExpression b, Dictionary<string, object?> localScope, EntityData data)
155 {
156 var left = EvaluateExpression(entity, b.Left, localScope, data);
157 if (left == null) return false;
158
159 bool leftBool = (bool)left;
160
161 if (b.Operator == "or")
162 {
163 if (leftBool) return true;
164 var right = EvaluateExpression(entity, b.Right, localScope, data);
165 return right != null && (bool)right;
166 }
167
168 if (b.Operator == "and")
169 {
170 if (!leftBool) return false;
171 var right = EvaluateExpression(entity, b.Right, localScope, data);
172 return right != null && (bool)right;
173 }
174
175 throw new Exception($"Unknown logic operator: {b.Operator}");
176 }
177
178 private static bool EvaluateUnary(Entity entity, UnaryLogicExpression u, Dictionary<string, object?> localScope, EntityData data)
179 {
180 var val = EvaluateExpression(entity, u.Inner, localScope, data);
181 if (val == null) return true;
182 return !(bool)val;
183 }

185 private static object? GetPoolProperty(Entity entity, PoolPropertyExpression p, EntityData data)
186 {
187 if (entity.Fields.TryGetValue(p.TargetName, out var obj) && obj is MorphynPool pool)
188 {
189 if (p.Property == "count") return (double)pool.Values.Count;
190 throw new Exception($"Property '{p.Property}' not supported for pools.");
191 }
192
193 if (data.Entities.TryGetValue(p.TargetName, out var externalEntity))
194 {
195 if (externalEntity.Fields.TryGetValue(p.Property, out var val))
196 {
197 return val;
198 }
199 throw new Exception($"Field '{p.Property}' not found in external entity '{p.TargetName}'");
200 }
201
202 throw new Exception($"Entity or Pool '{p.TargetName}' not found.");
203 }

205 private static object? GetFromPool(Entity entity, IndexAccessExpression idx, Dictionary<string, object?> localScope, EntityData data)
206 {
207 if (entity.Fields.TryGetValue(idx.TargetName, out var val) && val is MorphynPool pool)
208 {
209 var evalIndex = EvaluateExpression(entity, idx.IndexExpr, localScope, data);
210 if (evalIndex == null)
211 throw new Exception($"Index expression evaluated to null for pool '{idx.TargetName}'");
212
213 int index = Convert.ToInt32(evalIndex) - 1;
214
215 if (index >= 0 && index < pool.Values.Count)
216 return pool.Values[index];
217
218 throw new Exception($"Runtime Error: Index {index + 1} is out of bounds for pool '{idx.TargetName}'");
219 }
220 throw new Exception($"Target '{idx.TargetName}' is not a pool.");
221 }

223 private static object EvaluateBinary(Entity entity, BinaryExpression b, Dictionary<string, object?> localScope, EntityData data)
224 {
225 var leftObj = EvaluateExpression(entity, b.Left, localScope, data);
226 var rightObj = EvaluateExpression(entity, b.Right, localScope, data);
227
228 // Handle null comparisons for equality operators
229 if (b.Operator == "==" || b.Operator == "!=")
230 {
231 if (leftObj == null || rightObj == null)
232 {
233 return b.Operator == "==" ? Equals(leftObj, rightObj) : !Equals(leftObj, rightObj);
234 }
235 }
236 else
237 {
238 // For non-equality operators, null is not allowed
239 if (leftObj == null || rightObj == null)
240 {
241 throw new Exception($"Cannot perform operation '{b.Operator}' with null operand");
242 }
243 }
244
245 if (IsNumeric(leftObj) && IsNumeric(rightObj))
246 {
247 double l = Convert.ToDouble(leftObj, CultureInfo.InvariantCulture);
248 double r = Convert.ToDouble(rightObj, CultureInfo.InvariantCulture);
249
250 return b.Operator switch
251 {
252 "+" => l + r,
253 "-" => l - r,
254 "*" => l * r,
255 "/" => Math.Abs(r) > 1e-9 ? l / r : 0.0,
256 "%" => l % r,
257 ">" => l > r,
258 "<" => l < r,
259 ">=" => l >= r,
260 "<=" => l <= r,
261 "==" => Math.Abs(l - r) < 1e-7,
262 "!=" => Math.Abs(l - r) > 1e-7,
263 _ => throw new Exception($"Unknown operator: {b.Operator}")
264 };
265 }
266
267 return b.Operator switch
268 {
269 "==" => Equals(leftObj, rightObj),
270 "!=" => !Equals(leftObj, rightObj),
271 _ => throw new Exception($"Operator {b.Operator} not supported for these types")
272 };
273 }
274
275 private static bool IsNumeric(object? obj) =>
276 obj is sbyte || obj is byte || obj is short || obj is ushort ||
277 obj is int || obj is uint || obj is long || obj is ulong ||
278 obj is float || obj is double || obj is decimal;
279 }
